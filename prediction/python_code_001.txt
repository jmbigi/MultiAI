# This is a simple Python program
print("Hello, World!")

# Define a function
def greet(name):
    print("Hello, " + name)

# Call the function
greet("Alice")

# Use a for loop
for i in range(10):
    print(i)

# Define a class
class MyClass:
    def __init__(self, name):
        self.name = name

    def display_name(self):
        print(self.name)

# Create an object
obj = MyClass("Bob")
obj.display_name()
